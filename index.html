/* ========== Qlik lints & Secret scan ========== */
function lintQlik(fn, code){
  const warns=[];
  if(/(LOAD\s+[^\n]+\n)+\s*LOAD\s/i.test(code) && !/QUALIFY|UNQUALIFY/i.test(code)) warns.push('Consider QUALIFY/UNQUALIFY to avoid synthetic keys.');
  if(/JOIN\s/i.test(code) && !/KEY|CONCATENATE/i.test(code)) warns.push('Check join keys to avoid unintended cartesian joins.');
  if(/SECTION\s+ACCESS/i.test(code) && !/SECTION\s+APPLICATION/i.test(code)) warns.push('SECTION APPLICATION missing after SECTION ACCESS?');
  if(warns.length){ addMsg('assistant', `
Qlik lint for ${escapeHtml(fn)}:
• ${warns.map(escapeHtml).join('
• ')}
`); }
}
function scanSecrets(fn, code){
  const pats=[ /AKIA[0-9A-Z]{16}/, /-----BEGIN (RSA|EC|OPENSSH) PRIVATE KEY-----/, /ghp_[A-Za-z0-9]{36,}/, /xox[baprs]-[0-9A-Za-z-]{10,}/ ];
  if(pats.some(re=>re.test(code))){ addMsg('assistant', `Potential secret detected in ${escapeHtml(fn)} — review before committing.`); }
}
